Slide 2 / 14 🏭 Pengantar Factory Constructors
  • Apa itu? Constructor khusus yang tidak selalu membuat instance baru
  • Keyword: factory
  • Perbedaan dengan generative constructor:
    • Generative: Selalu membuat instance baru
    • Factory: Bisa mengembalikan instance yang sudah ada, atau instance subclass
// Generative constructor (selalu baru)
class Point {
    double x, y;
    Point(this.x, this.y);  // generative → selalu buat instance baru
}

// Factory constructor (bisa return instance existing)
class Logger {
    static final Logger _instance = Logger._internal();
    Logger._internal();
    
    factory Logger() {
        return _instance;  // mengembalikan instance yang sama (singleton)
    }
}
📌 Inti Factory Constructor: Memberikan kontrol penuh atas proses pembuatan objek – apakah membuat baru, mengambil dari cache, atau mengembalikan subclass.

class DatabaseConnection {
    // Private constructor
    DatabaseConnection._internal() {
        print('Creating database connection...');
    }
    
    // Static instance
    static final DatabaseConnection _instance = DatabaseConnection._internal();
    
    // Factory constructor
    factory DatabaseConnection() {
        return _instance;
    }
    
    void connect() {
        print('Connected to database');
    }
}

void main() {
    var db1 = DatabaseConnection();
    var db2 = DatabaseConnection();
    print(identical(db1, db2)); // true - instance yang sama
    db1.connect();
}

class ConnectionPool {
  final String host;
  final int port;

  // Cache berdasarkan parameter
  static final Map _cache = {};

  // Private constructor
  ConnectionPool._(this.host, this.port);

  // Factory dengan caching
  factory ConnectionPool(String host, int port) {
    var key = '$host:$port';

    // Return dari cache jika ada
    if (_cache.containsKey(key)) {
      print('Returning cached connection for $key');
      return _cache[key]!;
    }

    // Buat baru dan cache
    print('Creating new connection for $key');
    var connection = ConnectionPool._(host, port);
    _cache[key] = connection;
    return connection;
  }

  void query(String sql) {
    print('Executing: "$sql" on $host:$port');
  }
}

void main() {
  print('--- Creating conn1 (localhost:5432) ---');
  var conn1 = ConnectionPool('localhost', 5432);
  conn1.query('SELECT * FROM users;');
  print('');

  print('--- Creating conn2 (localhost:5432 - should be cached) ---');
  var conn2 = ConnectionPool('localhost', 5432); // Dari cache
  conn2.query('INSERT INTO products (name) VALUES ("Laptop");');
  print('');

  print('--- Creating conn3 (db.example.com:3306 - new connection) ---');
  var conn3 = ConnectionPool('db.example.com', 3306); // Baru
  conn3.query('UPDATE orders SET status="shipped" WHERE id=101;');
  print('');

  // Demonstrate that conn1 and conn2 are the same instance
  print('Are conn1 and conn2 the same instance? ${identical(conn1, conn2)}');
  // Demonstrate that conn1 and conn3 are different instances
  print('Are conn1 and conn3 the same instance? ${identical(conn1, conn3)}');
}
💡 Hasil: Memanggil ConnectionPool('localhost', 5432) dua kali akan mengembalikan instance yang sama dari cache.

abstract class Shape {
  void draw();

  factory Shape(String type) {
    switch (type.toLowerCase()) {
      case 'circle':
        return Circle();
      case 'square':
        return Square();
      case 'triangle':
        return Triangle();
      case 'rectangle':
        return Rectangle();
      default:
        print('Unknown shape: $type. Using Circle as default.');
        return Circle();
    }
  }
}

class Circle implements Shape {
  @override
  void draw() => print('Drawing circle');
}

class Square implements Shape {
  @override
  void draw() => print('Drawing square');
}

class Triangle implements Shape {
  @override
  void draw() => print('Drawing triangle');
}

class Rectangle implements Shape {
  @override
  void draw() => print('Drawing rectangle');
}

void main() {
  var circle = Shape('circle');
  var square = Shape('square');
  var triangle = Shape('triangle');
  var rectangle = Shape('rectangle');
  var unknownShape = Shape('pentagon');

  circle.draw();
  square.draw();
  triangle.draw();
  rectangle.draw();
  unknownShape.draw();
}

        return AppConfig._development();
      case 'prod':
        return AppConfig._production();
      case 'production':
        return AppConfig._production();
      case 'stage':
        return AppConfig._staging();
      case 'staging':
        return AppConfig._staging();
      default:
        throw ArgumentError('Unknown environment: $env');
    }
  }

    print('API URL: $apiUrl');
    print('Debug: $debugMode');

class User {

    final String email;

    User._({required this.id, required this.name, required this.email});

    // Factory dari JSON
    factory User.fromJson(Map json) {
    return User._(
    id: json['id']?.toString() ?? '',
    name: json['name']?.toString() ?? '',
    email: json['email']?.toString() ?? '',
    );
    }

    // Factory untuk guest user
    factory User.guest() {
    return User._(
    id: 'guest_${DateTime.now().millisecondsSinceEpoch}', 
    name: 'Guest', 
    email: 'guest@example.com', 
    );
    }

    // Factory dengan validation
    factory User.create(String name, String email) {
    if (name.isEmpty) {
    throw ArgumentError('Name cannot be empty');
    }
    if (!email.contains('@')) {
    throw ArgumentError('Invalid email');
    }

    return User._(
    id: 'user_${DateTime.now().millisecondsSinceEpoch}', 
    name: name, 
    email: email, 
    );
    }

    @override
    String toString() {
    return 'User(id: $id, name: $name, email: $email)';
    }
}

void main() {
    // Penggunaan

    'id': '123',

class Example {
    Example._() {}   // private constructor
    
    // Factory constructor
    factory Example.create() => Example._();
    
    // Static method
    static Example createStatic() => Example._();
}

void main() {
    var e1 = Example.create();        // Factory constructor
    var e2 = Example.createStatic();  // Static method
    
    print(identical(e1, e2)); // false (instance berbeda)
}

Slide 12 / 14 🎨 Factory untuk Immutable Objects dengan Cache (Color) 
//nama:Fajar Rio Dinata
//nim:251410025
//kelas:S12A
class Color {
    final int red;
    final int green;
    final int blue;
    
    // Cache immutable objects
    static final Map<String, Color> _cache = {};
    
    // Private constructor
    Color._(this.red, this.green, this.blue);
    
    // Factory dengan cache
    factory Color.rgb(int red, int green, int blue) {
        // Validasi
        if (red < 0 || red > 255 || green < 0 || green > 255 || blue < 0 || blue > 255) {
            throw ArgumentError('RGB values must be 0-255');
        }
        var key = '$red,$green,$blue';
        if (!_cache.containsKey(key)) {
            _cache[key] = Color._(red, green, blue);
        }
        return _cache[key];
    }
    
    // Named factories untuk warna umum
    factory Color.red() => Color.rgb(255, 0, 0);
    factory Color.green() => Color.rgb(0, 255, 0);
    factory Color.blue() => Color.rgb(0, 0, 255);
    factory Color.white() => Color.rgb(255, 255, 255);
    factory Color.black() => Color.rgb(0, 0, 0);
    
    @override
    String toString() => 'Color($red, $green, $blue)';
}
💡 Hasil: identical(Color.red(), Color.rgb(255,0,0)) → true (instance sama karena cache)

abstract class Document {
  void render();

  factory Document(String type, String content) {
    switch (type.toLowerCase()) {
      case 'pdf':
        return PDFDocument(content);
      case 'word':
        return WordDocument(content);
      case 'html':
        return HTMLDocument(content);
      case 'markdown':
        return MarkdownDocument(content);
      default:
        return PlainTextDocument(content);
    }
  }
}

class PDFDocument implements Document {
  final String content;
  PDFDocument(this.content);

  @override
  void render() {
    print('Rendering PDF: $content');
  }
}

class WordDocument implements Document {
  final String content;
  WordDocument(this.content);

  @override
  void render() {
    print('Rendering Word: $content');
  }
}

class HTMLDocument implements Document {
  final String content;
  HTMLDocument(this.content);

  @override
  void render() {
    print('Rendering HTML: $content');
  }
}

class MarkdownDocument implements Document {
  final String content;
  MarkdownDocument(this.content);

  @override
  void render() {
    print('Rendering Markdown: $content');
  }
}

class PlainTextDocument implements Document {
  final String content;
  PlainTextDocument(this.content);

  @override
  void render() {
    print('Rendering plain text: $content');
  }
}

void main() {
  // Contoh untuk pdf dan html
  var doc1 = Document('pdf', 'Hello PDF');
  var doc2 = Document('html', '
Hello
');

  doc1.render(); // Rendering PDF: Hello PDF
  doc2.render(); // Rendering HTML: 
Hello


  // Contoh untuk markdown, word, dan default (plain text)
  var doc3 = Document('markdown', '# This is a heading');
  var doc4 = Document('word', 'This is a Word document content.');
  var doc5 = Document('unknown type', 'This will be rendered as plain text.');

  doc3.render(); // Rendering Markdown: # This is a heading
  doc4.render(); // Rendering Word: This is a Word document content.
  doc5.render(); // Rendering plain text: This will be rendered as plain text.
}

Slide 14 / 14 📝 Ringkasan & Best Practices
Rangkuman:
  • Factory constructor: factory class.name()
  • Bisa return: instance yang ada, subclass, atau baru
  • Use cases: singleton, caching, object pools, subclass selection
Best Practices:
  • Gunakan factory untuk: Object pooling/caching, Singleton pattern, Complex object creation, Subclass selection
  • Gunakan generative constructor untuk: Simple object creation, Initialization lists, Required parameters
  • Validation: Lakukan validasi di factory sebelum membuat objek
